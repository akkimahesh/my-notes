#frontend
server {
    listen 80;
    server_name web.maheshakki.shop;

    root /var/www/monolithic/frontend/sceneary-app/build/web;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }
}

#backend
server {
    listen 80;
    server_name backend.maheshakki.shop;

    location / {
        proxy_pass http://127.0.0.1:8001;

        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;

        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}

#give permission
================
chown -R www-data:www-data /var/www/monolithic
chmod -R 755 /var/www/monolithic


#remove default
===============
rm -f /etc/nginx/sites-enabled/default

#enable the site our site
=========================
ln -s /etc/nginx/sites-available/maheshakki /etc/nginx/sites-enabled/

#restart nginx
==============
nginx -t
systemctl restart nginx
